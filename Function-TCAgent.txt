function Get-TCAgentList(){
        
    $agentAgentListUrl = "$teamCityHost/app/rest/agents"
    $agentresponse = Invoke-TCWebRequest -URL $agentAgentListUrl -Method Get

    $agents = ([xml]$agentresponse.Content).agents.agent 
    
    return $agents
}

function Get-TCAgentDetails($AgentId=$null, $AgentName=$null){
   
    $agents = Get-TCAgentList 
    
    if($AgentId){
        $agents = $agents | Where-Object { $_.Id -eq $AgentId }

    }elseif($AgentName)
    {
        $agents = $agents | Where-Object { $_.Name -eq $AgentName }
    }
    else{
        Write-Output "Please specify the Agent Id or Name"
    }
    
    $agentDtlsUrl = "$teamCityHost/app/rest/agents/id:$($agents.Id)"

    try{
         $TCAgentDetailsResponse = Invoke-TCWebRequest -URL $agentDtlsUrl -Method Get
    }
    catch{
        throw "Unable to find the build project!"
    }
    $TCAgentDetails = ([xml]$TCAgentDetailsResponse.Content).agent
    return $TCAgentDetails

}

function Get-TCAgentPoolList(){

    $agentpoollisturl = "$teamCityHost/app/rest/agentPools"
    $agentpoolresponse = Invoke-TCWebRequest -URL $agentpoollisturl -Method Get

    $agentpools = ([xml]$agentpoolresponse.Content).agentPools.agentPool
    return $AgentPools
}
